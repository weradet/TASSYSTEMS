ไดเรคทอรีเว็บได้ติดตั้งเฟรมเวิร์ก CodeIgniter 3.1.11 ไว้แล้ว

 

 

ไฟล์ที่ปรับแต่งไปแล้ว

 

ไฟล์ application/config/config.php

$config['base_url'] = 'http://localhost/cluster-demo';

$config['sess_save_path'] = '/tmp';

 

 

 

ไฟล์ application/config/autoload.php

$autoload['libraries'] = array('database', 'session');

$autoload['helper'] = array('url');

 

ไฟล์ application/config/database.php

$db['default'] = array(

    'dsn'   => '',

    'hostname' => 'db',

    'username' => 'cluster_demo',

    'password' => 'secret',

    'database' => 'db_cluster_demo',

    'dbdriver' => 'mysqli',

    'dbprefix' => '',

    'pconnect' => FALSE,

    'db_debug' => (ENVIRONMENT !== 'production'),

    'cache_on' => FALSE,

    'cachedir' => '',

    'char_set' => 'utf8',

    'dbcollat' => 'utf8_general_ci',

    'swap_pre' => '',

    'encrypt' => FALSE,

    'compress' => FALSE,

    'stricton' => FALSE,

    'failover' => array(),

    'save_queries' => TRUE

);

 

 

 

ข้อมูล MariaDB

ดูในไฟล์ mariadb/db.env

Note: การเก็บข้อมูลรหัสผ่านใดๆ ไว้ใน git เป็นสิ่งที่ไม่ควรกระทำ

 

ตารางข้อมูลจะถูกสร้างตามโครงสร้างตารางในไฟล์ mariadb/initdb/create-tables.sql

หากปรับโครงสร้างตารางแล้ว จะต้องลบไฟล์ในไดเรคทอรี mariadb/data/* ออกทั้งหมด
